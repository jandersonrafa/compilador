// Funcao media que retorna um real

funcao real media (inteiro a) {
  inteiro i
  inteiro soma
  inteiro quantidade
  real retorno

  soma := 0
  quantidade := 0
  
  para (i := 1; i <= a; i++) {  
    se (a%i == 0) {
      quantidade++
      soma := soma + i
    }
  }

  retorno := soma / quantidade
  retornar retorno
}
